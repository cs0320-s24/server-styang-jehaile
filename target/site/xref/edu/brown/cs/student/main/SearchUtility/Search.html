19  /** Class used to search a CSV file parsed using the CSVParser class. */
20  public class Search {
21  
22    private final CSVParser<List<String>> csvParser;
23  
24    /**
25     * Constructs a Search object using the specified file name to create a CSVParser.
26     *
27     * @param fileName The name of the CSV file to be searched.
28     * @param headersPresent A boolean indicating whether the CSV file has headers.
29     * @throws FileNotFoundException If the specified CSV file is not found.
30     * @throws AccessDeniedException If there is an attempt to access a file outside the 'data'
31     *     directory.
32     */
33    public Search(String fileName, Boolean headersPresent)
34        throws FileNotFoundException, AccessDeniedException {
35  
36      // Check security clearance during instantiation of Search object
37      String folderPath = "data";
38      Path absoluteFolderPath = Paths.get(folderPath).toAbsolutePath();
39      Path csvPath = Paths.get(absoluteFolderPath + "/" + fileName);
40  
41      File csvFile = new File(String.valueOf(csvPath));
42  
43      // Throws FileNotFoundException if file path does not exist
44      Reader reader = new FileReader(csvFile);
45  
46      // Check for injection (file could exist, but be injected outside the 'data' folder
47      if (String.valueOf(csvPath).contains("/../")) {
48        throw new AccessDeniedException(fileName);
49      }
50  
51      CreatorFromRow<List<String>> searchStrategy = new SearchStrategy();
52      this.csvParser = new CSVParser<>(reader, searchStrategy, headersPresent);
53    }
54  
55    /**
56     * Parses CSV stored in CSVParser field
57     *
58     * @throws IOException if an I/O error occurs while parsing the CSV file.
59     * @throws FactoryFailureException if an error occurs during CSVParser creation of List<String>
60     *     object from row.
61     * @throws MalformedRowsException if parser encounters rows of invalid size while parsing
62     */
63    public void parseCSV() throws IOException, FactoryFailureException, MalformedRowsException {
64      this.csvParser.parse();
65    }
66  
67    /**
68     * Searches the CSV file for rows containing specified value. The specified value must be a
69     * substring of one of the values in a row to be considered a 'match'.
70     *
71     * @param toSearch The value to search for in the CSV file.
72     * @return A list of rows that contain the specified value. If no matches are found, an empty list
73     *     will be returned.
74     */
75    public List<List<String>> searchCSV(String toSearch) {
76      List<List<String>> matches = new ArrayList<>();
77      // Search each row
78      for (List<String> row : this.csvParser.getCSVContents()) {
79        for (String value : row) {
80          if (value.toUpperCase().contains(toSearch.toUpperCase())) {
81            matches.add(row);
82            break; // One match is enough
83          }
84        }
85      }
86  
87      return matches;
88    }
89  
90    /**
91     * Searches the CSV file for rows containing the specified value in a specific header/column. The
92     * specified value must be a substring the value in the specified column of a row to be considered
93     * a 'match'.
94     *
95     * @param toSearch The value to search for in the CSV file.
96     * @param header The header/column in which to search for the value.
97     * @return A list of rows that contain the specified value in the specified header/column. If no
98     *     matches are found, an empty list will be returned.
99     * @throws NoSuchElementException if the specified header does not exist in the CSV file.
100    */
101   public List<List<String>> searchCSV(String toSearch, String header)
102       throws NoSuchElementException {
103     header = header.toUpperCase();
104     if (this.csvParser.getHeaderList().contains(header)) {
105       int index = this.csvParser.getHeaderList().indexOf(header);
106       return this.searchCSV(toSearch, index);
107     } else {
108       throw new NoSuchElementException();
109     }
110   }
111 
112   /**
113    * Searches the CSV file for rows containing the specified value in a specific column. The
114    * specified value must be a substring the value in the specified column of a row to be considered
115    * a 'match'.
116    *
117    * @param toSearch The value to search for in the CSV file.
118    * @param columnNumber The column number in which to search for the value.
119    * @return A list of rows that contain the specified value in the specified column. If no matches
120    *     are found, an empty list will be returned.
121    * @throws IndexOutOfBoundsException if the specified column number is out of bounds.
122    */
123   public List<List<String>> searchCSV(String toSearch, Integer columnNumber)
124       throws IndexOutOfBoundsException {
125     if (columnNumber < 0) {
126       throw new IndexOutOfBoundsException();
127     }
128     List<List<String>> matches = new ArrayList<>();
129     for (List<String> row : this.csvParser.getCSVContents()) {
130       if (columnNumber > row.size() - 1) {
131         throw new IndexOutOfBoundsException();
132       }
133       if (row.get(columnNumber).toUpperCase().contains(toSearch.toUpperCase())) {
134         matches.add(row);
135       }
136     }
137     return matches;
138   }
139 }
