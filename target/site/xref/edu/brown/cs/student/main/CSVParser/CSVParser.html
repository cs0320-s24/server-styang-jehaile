9   import java.util.Arrays;
10  import java.util.List;
11  import java.util.regex.Pattern;

13  /**
14   * Class that parses an inputted Reader object to List<T> based on an inputted CreatorFromRow<T>
15   * object
16   *
17   * @param <T> The type which each row of data is stored as in List<T> type csvContents field
18   */
19  public class CSVParser<T> {
20    private final Reader reader;
21    private final CreatorFromRow<T> strategyObject;
22    private final Boolean headers;
23    private ArrayList<String> headerList;
24    private List<T> csvContents;
25  
26    /**
27     * Constructor for CSVParser object.
28     *
29     * @param reader a Reader object to be parsed
30     * @param strategyObject a CreatorFromRow<T> object used to create object of type T from each row
31     *     of reader
32     * @param headers a Boolean indicating if headers are present in reader.
33     */
34    public CSVParser(Reader reader, CreatorFromRow<T> strategyObject, Boolean headers) {
35      this.reader = reader;
36      this.strategyObject = strategyObject;
37      this.headers = headers;
38    }
39  
40    /**
41     * Parses Reader file, assigning CSVParser's csvContents field a list of generic type T, where T
42     * indicates the type generated by the Class's strategyObject's create() method for each inputted
43     * row. If CSVParser's headers field is True, the parse() method will assign CSVParser's
44     * headerList field an ArrayList of String storing the first row of the Reader file. If the Reader
45     * file is empty, csvContents will be initiated as an empty list. If headers is True, headerList
46     * will also be initiated as an empty list, otherwise it will remain unassigned. Uses the length
47     * of the first line of CSVParser's reader field as the standard row size and skips invalid-sized
48     * rows while parsing, storing them in malformedRows. If malformedRows is not empty at the end of
49     * parsing each row, a MalformedRowsException will be thrown.
50     *
51     * @throws IOException parse() method throws IOException if encountering an error reading from
52     *     Reader in CSVParser's reader field.
53     * @throws FactoryFailureException parse() method throws FactoryFailureException if CreatorFromRow
54     *     object encounters an error when converting row to generic type T
55     * @throws MalformedRowsException parse() method throws MalformedRowsException after encountering
56     *     rows that are the invalid size. Exception is thrown after parsing is completed
57     */
58    public void parse() throws IOException, FactoryFailureException, MalformedRowsException {
59      BufferedReader bufferedReader = new BufferedReader(this.reader);
60      Pattern regexSplitCSVRow = Pattern.compile(",(?=([^\\\"]*\\\"[^\\\"]*\\\")*(?![^\\\"]*\\\"))");
61      ArrayList<List<String>> malformedRows = new ArrayList<>();
62      this.csvContents = new ArrayList<>();
63  
64      // read first line
65      String line = bufferedReader.readLine();
66  
67      // empty file leaves csvContents field as an empty list. headersList also empty.
68      if (line == null) {
69        if (headers) {
70          this.headerList = new ArrayList<String>();
71        }
72        return;
73      }
74  
75      // assign rowSize based on first line, regardless of if headers are present
76      int rowSize = Arrays.asList(regexSplitCSVRow.split(line)).size();
77  
78      // build headerList if headers are present
79      if (this.headers) {
80        String[] headersArray = regexSplitCSVRow.split(line);
81        ArrayList<String> headerListBuilder = new ArrayList<>();
82        for (String header : headersArray) {
83          headerListBuilder.add(header.toUpperCase());
84        }
85        this.headerList = headerListBuilder;
86  
87        line = bufferedReader.readLine();
88      }
89  
90      // while contents exist, parse each row
91      while (line != null) {
92        List<String> result = Arrays.asList(regexSplitCSVRow.split(line));
93        if (result.size() == rowSize) {
94          this.csvContents.add(strategyObject.create(result));
95        } else {
96          // if malformed rows exist, they are logged, but parsing continues (error thrown after)
97          malformedRows.add(result);
98        }
99        line = bufferedReader.readLine();
100     }
101 
102     bufferedReader.close();
103 
104     // inform that malformed rows exist and were not parsed
105     if (!malformedRows.isEmpty()) {
106       throw new MalformedRowsException(
107           "Some rows could not be parsed due to invalid size", malformedRows);
108     }
109   }
110 
111   /**
112    * Getter method for CSVContents to be called after parse()
113    *
114    * @return List<T> containing parsed file contents or null if not parsed yet
115    */
116   public List<T> getCSVContents() {
117     return this.csvContents;
118   }
119 
120   /**
121    * Getter method for headerList to be called after parse()
122    *
123    * @return List<String> containing headers in order found in reader field or null if not parsed
124    *     yet, or if headers field is False
125    */
126   public List<String> getHeaderList() {
127     return this.headerList;
128   }
129 }
